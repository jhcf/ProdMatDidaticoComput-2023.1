Programa_testa_menor:
        READ p1
        READ p2
        PUSH p1
        PUSH p2
        CALL SubRotinaMenorQue
        POP comparacao
        WRITE comparacao
        STOP
SubRotinaMenorQue:
	POP endereco_retorno
	POP n2
	POP n1
	ACC 1
	STORE um
inicia_comparacao:
	LOAD n1
	JNZ testa_n2
	JUMP testa_final_n2
testa_n2:
	LOAD n2
	JNZ subtrai_todos
	JUMP n1_maior
subtrai_todos:
	LOAD n1
	SUB um
	STORE n1
	LOAD n2
	SUB um
	STORE n2
	JUMP inicia_comparacao
testa_final_n2:
	LOAD n2
	JNZ n1_menor
	JUMP iguais
n1_menor:
	ACC 1
	STORE resultado
	JUMP retorna
iguais:
	ACC 0
	STORE resultado
	JUMP retorna
n1_maior:
	ACC 0
	STORE resultado
	JUMP retorna
retorna
	PUSH resultado
	PUSH endereco_retorno
	RET
